## Task 2: Fix timing oracle on ADMIN_SECRET and DASHBOARD_TOKEN (H-1)

**Files:**

Python `str.__eq__` short-circuits on the first mismatched byte, enabling timing side-channel attacks to recover secrets byte-by-byte. Use `hmac.compare_digest()` which is constant-time.

Add a new test file `tests/test_metrics.py`:

```python
"""Verify all metric names referenced in Prometheus alerts exist in metrics.py."""
import yaml
import re
from pathlib import Path

import prometheus_client

# Import metrics module to register all metrics
import app.utils.metrics  # noqa: F401


def _get_registered_metric_names():
    """Return all metric names registered with prometheus_client."""
    return set(prometheus_client.REGISTRY._names_to_collectors.keys())


def test_alert_metric_names_exist():
    """Every metric name in alerts.yaml must be registered in metrics.py."""
    alerts_path = Path("docker/prometheus/alerts.yaml")
    content = alerts_path.read_text()
    data = yaml.safe_load(content)

    registered = _get_registered_metric_names()

    # Extract metric names from expr fields (ritapi_* pattern)
    missing = []
    for group in data.get("groups", []):
        for rule in group.get("rules", []):
            expr = rule.get("expr", "")
            alert = rule.get("alert", "unknown")
            # Find all ritapi_* metric references (strip _total, _bucket, _count suffixes)
            refs = re.findall(r"ritapi_\w+", expr)
            for ref in refs:
                # prometheus_client stores base name without _total/_bucket
                base = re.sub(r"_(total|bucket|count|sum)$", "", ref)
                if base not in registered and ref not in registered:
                    missing.append(f"Alert '{alert}': metric '{ref}' not registered")

    assert not missing, "Missing metrics:\n" + "\n".join(missing)
```
